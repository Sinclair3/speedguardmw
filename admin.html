Supabase SQL — Create Tables
Run this SQL once in your Supabase SQL Editor to create the sessions tracking table: 
create table sg_sessions (
  id          bigint generated always as identity primary key,
  device_id   text not null,
  nickname    text,
  platform    text,
  app_version text,
  created_at  timestamptz default now()
);
alter table sg_sessions enable row level security;
create policy "Anyone can insert" on sg_sessions for insert with check (true);
create policy "Anyone can read"  on sg_sessions for select using (true);

-- Run this too for reviews:
create table sg_reviews (
  id          bigint generated always as identity primary key,
  nickname    text,
  device_id   text,
  rating      int not null,
  categories  text,
  review_text text,
  app_version text,
  platform    text,
  created_at  timestamptz default now()
);
alter table sg_reviews enable row level security;
create policy "Anyone can insert" on sg_reviews for insert with check (true);
create policy "Anyone can read"   on sg_reviews for select using (true);

-- Run this too for admin password:
create table sg_admin_config (
  id    bigint generated always as identity primary key,
  key   text unique not null,
  value text not null
);
alter table sg_admin_config enable row level security;
create policy "Anyone can read"   on sg_admin_config for select using (true);
create policy "Anyone can insert" on sg_admin_config for insert with check (true);
create policy "Anyone can update" on sg_admin_config for update using (true);
